16.Checking Alphabets

#include <iostream>

using namespace std;
int find(char a,char a1[]){
    for (int i=0;i<5;i++){
        if(a==a1[i]){
            return 1;
        }
        
    }
    return 0;
}

int main() {
    char a;
    cin>>a;
    char a1[]={'a','e','i','o','u'};
    
    if(find(a,a1)){
        cout<<"Vowel";
    }
    else if((a>='A' && a<='Z') || (a>='a' && a<='z') ){
        cout<<"Consonant";
    }
    else{
        cout<<"Not an alphabet";
    }
    
    
    return 0;
}

18.Online shopping

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

void minimum(int f,int s, int a){
    if(f<=s && f<=a){
        cout<<"Choose Flipkart";
    }
    else if(s<f && s<=a){
        cout<<"Choose Snapdeal";
    }
    else if(a<f && a<s){
        cout<<"Choose Amazon";
    }
}

int main() {
    float fp,fd,fs,sp,sd,ss,ap,ad,as;
    cin>>fp>>fd>>fs>>sp>>sd>>ss>>ap>>ad>>as;
    
    int f=int((fp-((fd/100)*fp)+fs));
    int s=int(sp-(sd/100)*sp+ss);
    int a=int(ap-(ad/100)*ap+as);
    
    cout<<"In Flipkart: Rs."<<f<<"\n"<<"In Snapdeal: Rs."<<s<<"\n"<<"In Amazon: Rs."<<a<<"\n";
    minimum(f,s,a);
    return 0;
}

22.Sum of digit till single digit

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int sod(long long int num){
    int sum=0;
    while(num!=0){
        int digit=num%10;
        num=num/10;
        sum=sum+digit;
    }
    return sum;
}

int main() {
    long long int num;
    cin>>num;
    
    int sum = sod(num);
    if(sum>9){
        sum=sod(sum);
    }
    
    cout<<sum;
    return 0;
}

26.Arrays 1D - Ascending order

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    int a;
    cin>>a;
    int a1[a];
    
    for(int i=0;i<a;i++){
        cin>>a1[i];
    }
    
    for(int i=0;i<a-1;i++){
        bool sorted=false;
        for(int j=0;j<a-i-1;j++){
            if(a1[j]>a1[j+1]){
                int temp=a1[j];
                a1[j]=a1[j+1];
                a1[j+1]=temp;
                sorted=true;
            }
        }
        if(sorted==false){
            break;
        }
    }
    cout<<"The Sorted array is:\n";
    for(int i=0;i<a;i++){
        cout<<a1[i]<<"\n";
    }
    return 0;
}


27.Arrays 1D - Array insertion

#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
void read(int size,int a1[]){
    for(int i=0;i<size;i++){
        cin>>a1[i];
    }
}

int main() {
    int a;
    cin>>a;
    int a1[a+1];
    read(a,a1);
    int pos;
    cin>>pos;
    if(pos<=0 || pos>a){
        cout<<"Invalid Input";
        return 0;
    }
    
    int ele;
    cin>>ele;
    
    for (int i = a-1; i >=pos-1; i--)
    {
        a1[i+1] = a1[i];
    }
    a1[pos-1] = ele;
    
    cout<<"Array after insertion is\n";
    for(int i=0;i<a+1;i++){
        cout<<a1[i]<<"\n";
    }
    return 0;
}


28.Arrays 2D - Transpose Matrix(in C)

#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#define MAX_SIZE 100
void readMatrix(int num_matrix[][MAX_SIZE],int size){
    for(int R=0; R < size; R++) {
        for(int C=0; C < size; C++) {
            scanf("%d",&num_matrix[R][C]);
        }
    }
}
void printTranspose(int num_matrix[][MAX_SIZE],int size){
    for(int R=0; R < size; R++) {
        for(int C=0; C < size; C++) {
            printf("%d ",num_matrix[R][C]);
        }
        printf("\n");
    }
    printf("Transpose matrix is:\n");
    for(int R=0; R < size; R++) {
        for(int C=0; C < size; C++) {
            printf("%d ",num_matrix[C][R]);
        }
        printf("\n");
    }
}
int main() {
    int num_matrix[MAX_SIZE][MAX_SIZE];
    int size;
    scanf("%d",&size);
    readMatrix(num_matrix,size);
    printTranspose(num_matrix,size);
    return 0;
}
